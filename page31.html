SQL — bases indispensables : DELETE

delete Commande DELETE

• Permet de supprimer des données d'une table

• Structure de base : DELETE FROM table WHERE condition

• Très dangereuse : sans WHERE, toutes les lignes sont supprimées !

code Exemples de DELETE

DELETE FROM messages WHERE id = 1;

Supprime le message avec l'ID 1

DELETE FROM messages WHERE auteur = 'Bob';

Supprime tous les messages de l'auteur 'Bob'

DELETE FROM messages WHERE date_creation < '2023-01-01';

Supprime les messages avant 2023

warning

ATTENTION : Oublier la clause WHERE supprime TOUTES les données de la table !

lightbulb Points clés
check_circle Toujours vérifier la clause WHERE avant d'exécuter
check_circle Faire une sauvegarde avant de supprimer des données

Exemple complet avec DELETE

-- Supprimer un message spécifique
DELETE FROM messages WHERE id = 1;

-- Supprimer tous les messages d'un auteur
DELETE FROM messages WHERE auteur = 'Alice';

-- Supprimer avec une condition complexe
DELETE FROM messages
WHERE date_creation < '2023-01-01'
AND auteur = 'Bob';

-- Supprimer les messages contenant un mot
DELETE FROM messages
WHERE message LIKE '%spam%';

-- Supprimer les messages anciens (plus de 30 jours)
DELETE FROM messages
WHERE date_creation < DATE_SUB(NOW(), INTERVAL 30 DAY);

-- DANGER: Supprimer TOUTES les données de la table
-- À utiliser avec une extrême prudence !
DELETE FROM messages;

-- Alternative plus sûre: Vider la table
TRUNCATE TABLE messages;
