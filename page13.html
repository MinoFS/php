Code complet du mini-projet

<?php
// Initialisation des variables
$messages = array();
$nom = "";
$prenom = "";
$message = "";
$erreur = "";

// Traitement du formulaire
if (isset($_POST['submit'])) {
  // Récupération des données
  $nom = isset($_POST['nom']) ? $_POST['nom'] : "";
  $prenom = isset($_POST['prenom']) ? $_POST['prenom'] : "";
  $message = isset($_POST['message']) ? $_POST['message'] : "";

  // Validation des champs
  if (empty($nom) || empty($prenom) || empty($message)) {
    $erreur = "Veuillez remplir tous les champs.";
  } else {
    // Ajout du message à la liste
    $nouveauMessage = array(
      'nom' => $nom,
      'prenom' => $prenom,
      'message' => $message,
      'date' => date('d/m/Y H:i:s')
    );
    $messages[] = $nouveauMessage;

    // Réinitialisation du formulaire
    $nom = "";
    $prenom = "";
    $message = "";
  }
}
?>

<!DOCTYPE html>
<html lang="fr">
<head>
  <meta charset="UTF-8">
  <title>Livre d'or</title>
  <style>
    body { font-family: Arial, sans-serif; margin: 0; padding: 20px; }
    form { margin-bottom: 30px; padding: 20px; background-color: #f5f5f5; border-radius: 8px; }
    input, textarea { width: 100%; padding: 8px; margin-bottom: 10px; border: 1px solid #ddd; border-radius: 4px; }
    button { background-color: #4285F4; color: white; padding: 10px 15px; border: none; border-radius: 4px; cursor: pointer; }
    .message { border: 1px solid #ddd; padding: 15px; margin-bottom: 15px; border-radius: 8px; }
    .message-header { font-weight: bold; margin-bottom: 5px; }
    .message-date { font-size: 0.8em; color: #666; margin-bottom: 10px; }
    .erreur { color: red; margin-bottom: 10px; }
  </style>
</head>
<body>
  <h1>Livre d'or</h1>

  <form method="POST" action="">
    <h2>Laissez un message</h2>

    <?php if (!empty($erreur)) { ?>
      <div class="erreur"><?= $erreur ?></div>
    <?php } ?>

    <label for="nom">Nom :</label>
    <input type="text" id="nom" name="nom" value="<?= $nom ?>">

    <label for="prenom">Prénom :</label>
    <input type="text" id="prenom" name="prenom" value="<?= $prenom ?>">

    <label for="message">Message :</label>
    <textarea id="message" name="message" rows="5"><?= $message ?></textarea>

    <button type="submit" name="submit">Envoyer</button>
  </form>

  <h2>Messages</h2>

  <?php
  // Affichage des messages
  if (empty($messages)) {
    echo "<p>Aucun message pour le moment.</p>";
  } else {
    foreach ($messages as $msg) {
      echo '<div class="message">';
      echo '<div class="message-header">' . $msg['prenom'] . ' ' . $msg['nom'] . '</div>';
      echo '<div class="message-date">' . $msg['date'] . '</div>';
      echo '<div>' . $msg['message'] . '</div>';
      echo '</div>';
    }
  }
  ?>
</body>
</html>
