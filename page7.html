• Permet de combiner du texte et des variables

• Essentiel pour créer des messages dynamiques

Exemples d'opérations et concaténation

<?php

// Opérations arithmétiques
$a = 10;
$b = 3;

$somme = $a + $b; // 13
$diff = $a - $b; // 7
$produit = $a * $b; // 30
$quotient = $a / $b; // 3.333...
$reste = $a % $b; // 1
$puissance = $a ** $b; // 1000

// Concaténation avec prénom et nom
$prenom = "Jean";
$nom = "Dupont";
$nomComplet = $prenom . " " . $nom;

// Affichage des résultats
echo "Bonjour " . $nomComplet . "!" . "<br>";
echo "Somme: " . $somme . "<br>";
echo "Produit: " . $produit . "<br>";

// Combinaison d'opérations et concaténation
$age = 25;
$ageDans10Ans = $age + 10;
echo $nomComplet . " aura " . $ageDans10Ans . " ans dans 10 ans."<br>;

?>
