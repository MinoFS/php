Tableaux

view_list Qu'est-ce qu'un tableau ?

• Structure de données contenant plusieurs valeurs

• Chaque valeur accessible via un index ou une clé

• PHP propose plusieurs types de tableaux

Exemples de tableaux

<?php

// Création d'un tableau simple (indexé)
$fruits = array("Pomme", "Banane", "Orange", "Fraise");

// Accès à un élément par son index
echo $fruits[0]; // Affiche "Pomme"
echo $fruits[2]; // Affiche "Orange"

// Ajout d'un élément à la fin
$fruits[] = "Ananas";

// Modification d'un élément
$fruits[1] = "Poire";

// Parcours avec une boucle foreach
foreach ($fruits as $fruit) {
  echo "$fruit<br>";
}

// Parcours avec une boucle for
$nombreFruits = count($fruits);
for ($i = 0; $i < $nombreFruits; $i++) {
  echo "Fruit n°" . ($i + 1) . ": " . $fruits[$i] . "<br>";
}

// Tableau associatif (avec clés personnalisées)
$etudiant = array(
  "nom" => "Dupont",
  "prenom" => "Jean",
  "age" => 22
);

// Accès par clé
echo "Nom: " . $etudiant["nom"] . "<br>";
echo "Prénom: " . $etudiant["prenom"] . "<br>";

?>
