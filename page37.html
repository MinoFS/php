Exemple de requête bien formatée

-- Requête SELECT bien formatée
SELECT
  id,
  nom,
  email,
  date_creation
FROM
  utilisateurs
WHERE
  age > 18
  AND
  statut = 'actif'
ORDER BY
  date_creation DESC
LIMIT
  10;
